\chapter{Fibonacci}
\section{Definizione}
\begin{defn}[Numeri di Fibonacci]\index{Fibonacci!definizione}
	\begin{align*}
		\Fib{1}=&1\\
		\Fib{2}=&1\\
		\Fib{n}=&\Fib{n-1}+\Fib{n-2}\quad n>2
	\end{align*}
\end{defn}
