\chapter{Fibonacci}
\section{Definizione}
\begin{defn}[Numeri di Fibonacci]\index{Fibonacci!definizione}
	\begin{align}
		\Fib{1}=&1\notag\\
		\Fib{2}=&1\notag\\
		\Fib{n}=&\Fib{n-1}+\Fib{n-2}\quad n>2\label{eqn:Fibodef}
	\end{align}
\end{defn}
